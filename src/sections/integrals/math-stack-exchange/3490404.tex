\begin{problem}[pytanie 3490404]
    \label{stack_3490404}%
    Niech $f(x) = x \log (1 - \sin x) / \sin x$.
    Udowodnić, że
    \begin{align}
        2\int_0^{\pi/2} f(x) \,\mathrm{d}x =
        \int_{\pi/2}^\pi\ f(x) \,\mathrm{d}x,
    \end{align}
    bez uprzedniego znajdowania funkcji pierwotnej $f$.
\end{problem}
