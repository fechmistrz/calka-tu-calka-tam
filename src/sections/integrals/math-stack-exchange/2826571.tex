\begin{problem}[pytanie 2826571]
    \label{stack_2826571}%
    Niech $(m, n)$ oznacza największy wspólny dzielnik liczb $m, n$.
    Wtedy
    \begin{equation}
        \int_0^{\pi/2} \log \lvert \sin(mx) \rvert \cdot \log \lvert\sin(nx)\rvert \, dx = \frac{\pi^3}{24} \frac{(m,n)^2}{mn}+\frac{\pi\log (2)^2}{2}.
    \end{equation}
\end{problem}
