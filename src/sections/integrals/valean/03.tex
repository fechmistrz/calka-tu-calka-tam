\begin{problem_with_solution}[1.3]
    \label{valean_1_3}%
    Rozpatrujemy rodzinę całek
    \begin{equation}
        \int_0^1 x^{n-1} \log^k (1-x) \,\mathrm{d} x =: I(k).
    \end{equation}
    Mamy:
    \begin{align}
        I(1) & = - \frac{H_n}{n} \\
        I(2) & = \frac{H_n^2 + H_n^{(2)}}{n} \\
        I(3) & = - \frac{H_n^3 + 3H_nH_n^{(2)} + 2H_n^{(3)}}{n} \\
        I(4) & = \frac{H_n^4 + 6H_n^2 H_n^{(2)} + 8H_nH_n^{(3)} + 3(H_n^{(2)})^2 + 6H_n^{(4)}}{n},
    \end{align}
    gdzie $H_{n}^{(m)} = 1 + 1/2^m + \ldots + 1/n^m$ oznacza $n$-tą uogólnioną liczbę harmoniczną.
\end{problem_with_solution}
