\begin{problem_with_solution}[1.2]
    \label{valean_1_2}%
    Niech $m, n$ będą liczbami naturalnymi.
    Wtedy
    \begin{equation}
        \int_0^1 x^m \log^n x \,\mathrm{d} x = \frac{(-1)^n \cdot n!}{(m+1)^{n+1}}.
    \end{equation}
\end{problem_with_solution}
