    \begin{equation}
		\int_1^\infty \frac{\mathrm{d}x}{x^3 + x} = \log \sqrt {2}.
    \end{equation}

% SOLUTION
\textbf{Problem \ref{nahin_page_225}}.
Rozkładamy całkowaną funkcję na sumę
\begin{equation}
	\frac 1 x - \frac{1}{2(x - i)} - \frac{1}{2(x+i)}
\end{equation}
tak jak Nahin \cite[s. 225]{nahin15}.
